    private void addBall(String color) {
        for (int i = 0; i < 2; i++) {
            ballSlots[i] = ballSlots[i + 1];
        }
        ballSlots[2] = color;
    }
    private void rotateForShooting() {




        if (detectedBalls == 3) {

            String temp = ballSlots[0];
            ballSlots[0] = ballSlots[1];
            ballSlots[1] = ballSlots[2];
            ballSlots[2] = temp;
        } else if (detectedBalls == 2) {

            ballSlots[0] = ballSlots[2];
            ballSlots[1] = ballSlots[1];
            ballSlots[2] = null;
        } else if (detectedBalls == 1) {

            ballSlots[0] = ballSlots[2];
            ballSlots[1] = null;
            ballSlots[2] = null;
        }


        if (ShootingPattern.enablePatternMatching && detectedBalls > 0) {
            int maxCycles = 3;
            boolean matches = false;
            for (int cycle = 0; cycle < maxCycles; cycle++) {
                matches = true;
                for (int i = 0; i < detectedBalls; i++) {
                    if (ballSlots[i] != null && !ballSlots[i].equals("unknown") && !ballSlots[i].equals(currentPattern[i].trim())) {
                        matches = false;
                        break;
                    }
                }
                if (matches) break;


                String temp = ballSlots[0];
                for (int i = 0; i < 2; i++) {
                    ballSlots[i] = ballSlots[i + 1];
                }
                ballSlots[2] = temp;
                spindexerTargetPosition += (int) Spindexer.stepTicks;
            }

            if (!matches) {
                gamepad1.rumble(0.8, 0.8, 400);
            }
        }
    }
    private void clearBalls() {
        for (int i = 0; i < ballSlots.length; i++) {
            ballSlots[i] = null;
        }
        detectedBalls = 0;
        ballDetectedLastLoop = false;
    }
    private String formatBallSlots() {
        StringBuilder sb = new StringBuilder("[");
        for (int i = 0; i < ballSlots.length; i++) {
            if (ballSlots[i] == null) {
                sb.append("empty");
            } else if (ballSlots[i].equals("unknown")) {
                sb.append("???");
            } else {
                sb.append(ballSlots[i]);
            }
            if (i < ballSlots.length - 1) {
                sb.append(", ");
            }
        }
        sb.append("]");
        return sb.toString();
    }
    private String formatShootingOrder() {


        if (detectedBalls == 0) {
            return "No balls to shoot";
        }
        StringBuilder sb = new StringBuilder();
        sb.append("Order: ");
        for (int i = 0; i < detectedBalls; i++) {
            if (ballSlots[i] != null) {
                if (ballSlots[i].equals("unknown")) {
                    sb.append("???");
                } else {
                    sb.append(ballSlots[i]);
                }
            } else {
                sb.append("?");
            }
            if (i < detectedBalls - 1) {
                sb.append(" → ");
            }
        }
        return sb.toString();
    }
    private String detectColor() {

        int red1 = colorSensor.red();
        int green1 = colorSensor.green();
        int blue1 = colorSensor.blue();


        int red2 = colorSensor2.red();
        int green2 = colorSensor2.green();
        int blue2 = colorSensor2.blue();


        boolean purple1 = red1 >= PurpleSensor1.redMin && green1 >= PurpleSensor1.greenMin && blue1 >= PurpleSensor1.blueMin;
        boolean green1Detected = red1 >= GreenSensor1.redMin && green1 >= GreenSensor1.greenMin && blue1 >= GreenSensor1.blueMin;


        boolean purple2 = red2 >= PurpleSensor2.redMin && green2 >= PurpleSensor2.greenMin && blue2 >= PurpleSensor2.blueMin;
        boolean green2Detected = red2 >= GreenSensor2.redMin && green2 >= GreenSensor2.greenMin && blue2 >= GreenSensor2.blueMin;


        if (purple1 || purple2) return "purple";
        if (green1Detected || green2Detected) return "green";
        return "none";
    }

    private void handleTransfer() {
        if (!transferActive) return;

        int spindexerCurrentPos = spindexer.getCurrentPosition();
        double currentServoPos = servo.getPosition();


        if (Transfer.enableTransferTimeout &&
                (timer.milliseconds() - transferStageStartTime) > Transfer.transferStageTimeoutMs) {

            transferActive = false;
            shootingMode = false;
            spindexerTargetPosition -= (int) Spindexer.shootingModeOffset;
            servoPosition = ServoFlicker.minPosition;
            clearBalls();
            gamepad1.rumble(1.0, 1.0, 300);
            telemetry.addData("ERROR", "Transfer timeout!");
            return;
        }

        switch (transferStage) {
            case 0:
                double targetMax = ServoFlicker.maxPosition;
                servo.setPosition(targetMax);
                if (Math.abs(currentServoPos - targetMax) <= ServoFlicker.positionTolerance) {
                    if (!positionReached) {
                        positionReached = true;
                        if (timerDone) {
                            startTime = timer.milliseconds();
                            timerDone = false;
                        }
                    }
                    if (timer.milliseconds() >= startTime + ServoFlicker.flickUpWaitMs) {
                        transferStage = 1;
                        transferStageStartTime = timer.milliseconds();
                        timerDone = true;
                        positionReached = false;
                    }
                } else {
                    positionReached = false;
                }
                break;

            case 1:
                double targetMin = ServoFlicker.minPosition;
                servo.setPosition(targetMin);
                if (Math.abs(currentServoPos - targetMin) <= ServoFlicker.positionTolerance) {
                    if (!positionReached) {
                        positionReached = true;
                        if (timerDone) {
                            startTime = timer.milliseconds();
                            timerDone = false;
                        }
                    }
                    if (timer.milliseconds() >= startTime + ServoFlicker.flickDownWaitMs) {
                        flickCount++;


                        for (int i = 0; i < 2; i++) {
                            ballSlots[i] = ballSlots[i + 1];
                        }
                        ballSlots[2] = null;
                        detectedBalls--;

                        if (flickCount >= originalDetectedBalls) {
                            transferActive = false;
                            shootingMode = false;
                            spindexerTargetPosition -= (int) Spindexer.shootingModeOffset;
                            servoPosition = targetMin;
                            clearBalls();

                            spindexerAlignPending = true;


                            if (ShootingPattern.enablePatternCycling && originalDetectedBalls == 3) {

                                String temp = currentPattern[2];
                                currentPattern[2] = currentPattern[1];
                                currentPattern[1] = currentPattern[0];
                                currentPattern[0] = temp;
                                gamepad1.rumble(0.4, 0.4, 150);
                            }

                            if (Intake.autoPauseAfterShoot) {
                                intakeToggled = false;
                                gamepad1.rumble(0.3, 0.3, 200);
                            }
                            timerDone = true;
                            positionReached = false;
                        } else {
                            transferStage = 2;
                            transferStageStartTime = timer.milliseconds();
                            timerDone = true;
                            positionReached = false;
                        }
                    }
                } else {
                    positionReached = false;
                }
                break;

            case 2:
                spindexerTargetPosition += (int) Spindexer.stepTicks;
                transferStage = 3;
                transferStageStartTime = timer.milliseconds();
                positionReached = false;
                break;

            case 3:
                if (spindexerRetryAttempts >= Transfer.maxJamRetries) {
                    transferActive = false;
                    shootingMode = false;
                    spindexerTargetPosition -= (int) Spindexer.shootingModeOffset;
                    servoPosition = ServoFlicker.minPosition;
                    clearBalls();
                    spindexerRetryAttempts = 0;
                    spindexerLastRetryTime = 0;
                    intakeToggled = true;
                    gamepad1.rumble(1.0, 1.0, 500);
                    gamepad2.rumble(1.0, 1.0, 500);
                    telemetry.addData("TRANSFER", "ABORTED - JAM");
                    break;
                }

                double activeSpindexerError = spindexerTargetPosition - spindexerCurrentPos;
                telemetry.addData("Spindexer Transfer Error", (int) activeSpindexerError);
                boolean advanceReached = Math.abs(activeSpindexerError) <= Transfer.advanceTolerance;

                if (advanceReached) {
                    alignSpindexerToStep(spindexerCurrentPos);
                    transferStage = 4;
                    transferStageStartTime = timer.milliseconds();
                    positionReached = false;
                }
                break;

            case 4:

                if (!positionReached) {
                    positionReached = true;
                    if (timerDone) {
                        startTime = timer.milliseconds();
                        timerDone = false;
                    }
                }
                if (timer.milliseconds() >= startTime + ServoFlicker.postSpinWaitMs) {
                    transferStage = 0;
                    transferStageStartTime = timer.milliseconds();
                    timerDone = true;
                    positionReached = false;
                }
                break;
        }
    }

    private void handleFlywheel(double dt) {
        double velocity = motor.getVelocity();
        double error = Flywheel.targetVelocity - velocity;

        integralSum += error * dt;
        integralSum = Math.max(-Flywheel.integralMax, Math.min(Flywheel.integralMax, integralSum));

        double derivative = (error - lastError) / dt;
        double feedForward = Flywheel.kF * Flywheel.targetVelocity;

        double output = (Flywheel.kP * error) + (Flywheel.kI * integralSum) + (Flywheel.kD * derivative) + feedForward;

        output = Math.max(-1.0, Math.min(1.0, output));
        motor.setPower(output);
        lastError = error;

        if (Flywheel.targetVelocity == 0) {
            integralSum = 0;
            lastError = 0;
            motor.setPower(0);
            shooterRumbled = false;
        }

        if (Flywheel.targetVelocity > 0 && Math.abs(error) <= 50 && !shooterRumbled) {
            gamepad1.rumble(1.0, 1.0, 1000);
            shooterRumbled = true;
        }
    }

    private void handleTurret(double dt) {
        boolean currentG1DpadUp = gamepad1.dpad_up;
        if (currentG1DpadUp && !lastG1Dpad_up) {
            Tracking.enableTracking = !Tracking.enableTracking;
        }
        lastG1Dpad_up = currentG1DpadUp;

        boolean currentG2Circle = gamepad2.circle;
        if (currentG2Circle && !lastG2Circle) {
            Tracking.enableTracking = !Tracking.enableTracking;
        }
        lastG2Circle = currentG2Circle;


        if (Turntable.enableWraparound) {
            int currentTurretPos = turntable.getCurrentPosition();


            if (!turretWraparoundActive) {
                if (currentTurretPos > Turntable.maxTicks) {

                    turretWraparoundActive = true;
                    turretWraparoundTarget = Turntable.minTicks;
                    turretWraparoundIntegral = 0;
                    turretWraparoundLastError = 0;
                    turretWraparoundStartTime = timer.seconds();
                    gamepad1.rumble(0.3, 0.3, 150);
                } else if (currentTurretPos < Turntable.minTicks) {

                    turretWraparoundActive = true;
                    turretWraparoundTarget = Turntable.maxTicks;
                    turretWraparoundIntegral = 0;
                    turretWraparoundLastError = 0;
                    turretWraparoundStartTime = timer.seconds();
                    gamepad1.rumble(0.3, 0.3, 150);
                }
            }


            if (turretWraparoundActive) {
                double wraparoundElapsed = timer.seconds() - turretWraparoundStartTime;

                boolean emergencyCancel = (gamepad1.left_bumper && gamepad1.right_bumper) ||
                        (gamepad2.left_bumper && gamepad2.right_bumper);

                if (emergencyCancel || wraparoundElapsed > Turntable.wraparoundTimeoutSec) {
                    turretWraparoundActive = false;
                    turretWraparoundIntegral = 0;
                    turretWraparoundLastError = 0;
                    turntable.setPower(0);
                    gamepad1.rumble(1.0, 1.0, 300);
                    gamepad2.rumble(1.0, 1.0, 300);
                    if (wraparoundElapsed > Turntable.wraparoundTimeoutSec) {
                        Turntable.enableWraparound = false;
                        telemetry.addData("WRAPAROUND", "TIMEOUT - MANUAL MODE");
                    }
                } else {
                    double wraparoundError = turretWraparoundTarget - currentTurretPos;


                    if (Math.abs(wraparoundError) <= Turntable.wraparoundTolerance) {
                        turretWraparoundActive = false;
                        turretWraparoundIntegral = 0;
                        turretWraparoundLastError = 0;
                        turntable.setPower(0);
                        gamepad1.rumble(0.5, 0.5, 200);
                    } else {

                        turretWraparoundIntegral += wraparoundError * dt;
                        turretWraparoundIntegral = Math.max(-Turntable.wraparoundIntegralMax,
                                Math.min(Turntable.wraparoundIntegralMax,
                                        turretWraparoundIntegral));


                        double wraparoundDerivative = 0;
                        if (dt >= 1e-3) {
                            wraparoundDerivative = (wraparoundError - turretWraparoundLastError) / dt;
                        }

                        double wraparoundOutput = (Turntable.wraparoundKP * wraparoundError)
                                + (Turntable.wraparoundKI * turretWraparoundIntegral)
                                + (Turntable.wraparoundKD * wraparoundDerivative);

                        wraparoundOutput = Math.max(-Turntable.wraparoundMaxPower,
                                Math.min(Turntable.wraparoundMaxPower, wraparoundOutput));

                        turntable.setPower(wraparoundOutput);
                        turretWraparoundLastError = wraparoundError;
                    }
                }
            }
        }


        if (!turretWraparoundActive) {
            if (Tracking.enableTracking) {
                LLResult result = limelight.getLatestResult();

                if (result != null && result.isValid()) {
                    double tx = result.getTx();

                    if (tx < Tracking.minTx || tx > Tracking.maxTx) {
                        double trackingError = -tx;

                        if (trackingLastError != 0 && Math.signum(trackingError) != Math.signum(trackingLastError)) {
                            trackingIntegralSum = 0;
                        }
                        trackingIntegralSum += trackingError * dt;
                        trackingIntegralSum = Math.max(-Tracking.integralMax, Math.min(Tracking.integralMax, trackingIntegralSum));
                        double trackingDerivative = (trackingError - trackingLastError) / dt;
                        double trackingOutput = (Tracking.kP * trackingError) + (Tracking.kI * trackingIntegralSum) + (Tracking.kD * trackingDerivative);
                        trackingOutput = Math.max(-Tracking.turretPower, Math.min(Tracking.turretPower, trackingOutput));

                        turntable.setPower(-trackingOutput);
                        trackingLastError = trackingError;

                    } else {
                        turntable.setPower(0);
                        trackingIntegralSum = 0;
                        trackingLastError = 0;
                    }
                } else {
                    turntable.setPower(0);
                    trackingIntegralSum = 0;
                    trackingLastError = 0;
                }
            } else {
                if (gamepad1.right_bumper || gamepad2.right_bumper) {
                    turntable.setPower(Turntable.manualPower);
                } else if (gamepad1.left_bumper || gamepad2.left_bumper) {
                    turntable.setPower(-Turntable.manualPower);
                } else {
                    turntable.setPower(0);
                }

                trackingIntegralSum = 0;
                trackingLastError = 0;
            }
        } else {

            trackingIntegralSum = 0;
            trackingLastError = 0;
        }
    }

    private void handleServo(double dt) {
        if (!transferActive) {
            if (gamepad1.right_trigger > 0.1) {
                servoPosition += ServoFlicker.servoSpeed * gamepad1.right_trigger * dt;
            }
            if (gamepad1.left_trigger > 0.1) {
                servoPosition -= ServoFlicker.servoSpeed * gamepad1.left_trigger * dt;
            }
        }

        if (gamepad2.right_trigger > 0.1) {
            servoPosition += ServoFlicker.servoSpeed * gamepad2.right_trigger * dt;
        }
        if (gamepad2.left_trigger > 0.1) {
            servoPosition -= ServoFlicker.servoSpeed * gamepad2.left_trigger * dt;
        }

        servoPosition = Math.max(ServoFlicker.minPosition, Math.min(ServoFlicker.maxPosition, servoPosition));

        if (!transferActive) {
            servo.setPosition(servoPosition);
        }
    }

    private void handleShootingControls() {
        boolean currentTriangle = gamepad1.triangle;
        if (currentTriangle && !lastg1Triangle && !shootingMode && !transferActive) {
            alignSpindexerToStep(spindexer.getCurrentPosition());
            shootingMode = true;
            rotateForShooting();
            spindexerTargetPosition += (int) Spindexer.shootingModeOffset;
        }
        lastg1Triangle = currentTriangle;

        boolean currentCircle = gamepad1.circle;
        if (currentCircle && !lastG1Circle && shootingMode && !transferActive) {
            if (Transfer.preventZeroBallTransfer && (originalDetectedBalls == 0 || detectedBalls == 0)) {

                gamepad1.rumble(0.5, 0.5, 200);
            } else {
                transferActive = true;
                transferStage = 0;
                flickCount = 0;
                originalDetectedBalls = detectedBalls;
                timerDone = true;
                transferStageStartTime = timer.milliseconds();
            }
        }
        lastG1Circle = currentCircle;

        boolean g2DpadDown = gamepad2.dpad_down;
        if (g2DpadDown && !lastDpadDown && !transferActive) {
            spindexerTargetPosition -= (int) Spindexer.stepTicks;
            manualSpinUsed = true;

            if (detectedBalls > 0) {
                for (int i = 0; i < 2; i++) {
                    ballSlots[i] = ballSlots[i + 1];
                }
                ballSlots[2] = null;
                detectedBalls--;

                if (shootingMode && detectedBalls < 3) {
                    shootingMode = false;
                    spindexerTargetPosition -= (int) Spindexer.shootingModeOffset;
                    spindexerAlignPending = true;
                }
            }

            ballDetectedLastLoop = false;
            gamepad2.rumble(0.5, 0.5, 300);
        }
        lastDpadDown = g2DpadDown;

        boolean g2Triangle = gamepad2.triangle;
        if (g2Triangle && !lastG2Triangle && !shootingMode && !transferActive) {
            alignSpindexerToStep(spindexer.getCurrentPosition());
            shootingMode = true;
            rotateForShooting();
            spindexerTargetPosition += (int) Spindexer.shootingModeOffset;
        }
        lastG2Triangle = g2Triangle;

        boolean currentG2DpadUp = gamepad2.dpad_up;
        if (currentG2DpadUp && !lastG2DpadUp) {
            ShootingPattern.enablePatternMatching = !ShootingPattern.enablePatternMatching;
            gamepad2.rumble(ShootingPattern.enablePatternMatching ? 0.5 : 0.3, ShootingPattern.enablePatternMatching ? 0.5 : 0.3, 200);
        }
        lastG2DpadUp = currentG2DpadUp;

        boolean g2Square = gamepad2.square;
        if (g2Square && !lastG2Square && !shootingMode && !transferActive) {
            if (detectedBalls < 3) {
                addBall("unknown");
                detectedBalls++;

                if (detectedBalls == 3) {
                    if (Transfer.manualUnknownBallsAutoShoot) {
                        shootingMode = true;
                        rotateForShooting();
                        spindexerTargetPosition += (int) Spindexer.shootingModeOffset;
                        gamepad2.rumble(1.0, 1.0, 500);
                    } else {

                        spindexerTargetPosition += (int) Spindexer.stepTicks;
                        gamepad2.rumble(0.5, 0.5, 200);
                    }
                } else {
                    spindexerTargetPosition += (int) Spindexer.stepTicks;
                }
            } else {
                spindexerTargetPosition += (int) Spindexer.stepTicks;
            }
            manualSpinUsed = true;
        }
        lastG2Square = g2Square;

        boolean g1left = gamepad1.dpad_left;
        if (g1left && !lastDpadLeft && !transferActive) {
            spindexerTargetPosition -= (int) Spindexer.stepTicks;
            manualSpinUsed = true;
        }
        lastDpadLeft = g1left;

        boolean currentSquareState = gamepad1.square;
        if (currentSquareState && !lastSquareState && !shootingMode && !transferActive) {
            if (detectedBalls < 3) {
                addBall("unknown");
                detectedBalls++;

                if (detectedBalls == 3) {
                    if (Transfer.manualUnknownBallsAutoShoot) {
                        shootingMode = true;
                        rotateForShooting();
                        spindexerTargetPosition += (int) Spindexer.shootingModeOffset;
                        gamepad1.rumble(1.0, 1.0, 500);
                    } else {

                        spindexerTargetPosition += (int) Spindexer.stepTicks;
                        gamepad1.rumble(0.5, 0.5, 200);
                    }
                } else {
                    spindexerTargetPosition += (int) Spindexer.stepTicks;
                }
            } else {
                spindexerTargetPosition += (int) Spindexer.stepTicks;
            }
            manualSpinUsed = true;
        }
        lastSquareState = currentSquareState;
    }

    private void handleSpindexer(double dt) {
        int spindexerCurrentPos = spindexer.getCurrentPosition();
        double spindexerPosError = spindexerTargetPosition - spindexerCurrentPos;
        double currentTime = timer.milliseconds() / 1000.0;

        boolean isMoving = Math.abs(spindexerPosError) > Spindexer.positionTolerance;
        if (isMoving) {
            if (spindexerMoveStartTime == 0) {
                spindexerMoveStartTime = currentTime;
                spindexerRetryOriginalPos = spindexerTargetPosition;
            }
            double elapsed = currentTime - spindexerMoveStartTime;

            if (elapsed > Transfer.antiJamTimeoutSec && Math.abs(spindexerPosError) > Transfer.antiJamToleranceTicks) {
                double absoluteTime = timer.milliseconds() / 1000.0;
                if (spindexerLastRetryTime == 0 || (absoluteTime - spindexerLastRetryTime) > 0.5) {
                    spindexerRetryAttempts++;
                    spindexerLastRetryTime = absoluteTime;
                    spindexerInRetrySequence = true;
                    if (spindexerRetryAttempts >= Transfer.maxJamRetries) {
                        spindexer.setPower(0);
                        spindexerTargetPosition = spindexerCurrentPos;
                        gamepad1.rumble(1.0, 0, 500);
                        spindexerMoveStartTime = 0;
                        spindexerRetryAttempts = 0;
                        spindexerLastRetryTime = 0;
                        spindexerInRetrySequence = false;
                        return;//im losing my fucking mind fuck you fuck you fuck you fuck you
                    }
                    if (spindexerRetryAttempts % 2 == 1) {
                        spindexerRetryDirection = -1;
                        spindexerTargetPosition = spindexerCurrentPos - 30;
                    } else {
                        spindexerRetryDirection = 1;
                        spindexerTargetPosition = (int) Math.round(spindexerRetryOriginalPos);
                    }
                    spindexerIntegralSum = 0;
                    spindexerLastError = 0;
                    gamepad2.rumble(0.5, 0.5, 200);
                    telemetry.addData("SPINDEXER", "JAM RETRY #" + spindexerRetryAttempts + " -> " + (spindexerRetryDirection < 0 ? "BACK" : "FORWARD"));
                }
            }

            spindexerIntegralSum += spindexerPosError * dt;
            spindexerIntegralSum = Math.max(-Spindexer.integralMax, Math.min(Spindexer.integralMax, spindexerIntegralSum));
            double spindexerDerivative = (spindexerPosError - spindexerLastError) / dt;
            double spindexerOutput = (Spindexer.kP * spindexerPosError) + (Spindexer.kI * spindexerIntegralSum) + (Spindexer.kD * spindexerDerivative);
            spindexerOutput *= spindexerRetryDirection;
            spindexerOutput = Math.max(-Spindexer.maxPower, Math.min(Spindexer.maxPower, spindexerOutput));
            spindexer.setPower(spindexerOutput);
            spindexerLastError = spindexerPosError;
        } else {
            spindexer.setPower(0);
            spindexerIntegralSum = 0;
            spindexerLastError = 0;

            if (spindexerInRetrySequence) {
                boolean reachedOriginalTarget = Math.abs(spindexerCurrentPos - spindexerRetryOriginalPos) <= Spindexer.positionTolerance;
                if (reachedOriginalTarget) {
                    spindexerMoveStartTime = 0;
                    spindexerRetryAttempts = 0;
                    spindexerRetryDirection = 1.0;
                    spindexerLastRetryTime = 0;
                    spindexerInRetrySequence = false;
                } else {
                    double absoluteTime = timer.milliseconds() / 1000.0;
                    if (spindexerRetryAttempts % 2 == 1 && (absoluteTime - spindexerLastRetryTime) > 0.5) {
                        if (spindexerRetryAttempts + 1 >= Transfer.maxJamRetries) {
                            spindexer.setPower(0);
                            spindexerTargetPosition = spindexerCurrentPos;
                            spindexerMoveStartTime = 0;
                            spindexerRetryAttempts = 0;
                            spindexerRetryDirection = 1.0;
                            spindexerLastRetryTime = 0;
                            spindexerInRetrySequence = false;
                            gamepad1.rumble(1.0, 0, 500);
                            telemetry.addData("SPINDEXER", "MAX RETRIES - FAILED FUCKKKKKKK");
                        } else {
                            spindexerRetryAttempts++;
                            spindexerLastRetryTime = absoluteTime;
                            spindexerRetryDirection = 1;
                            spindexerTargetPosition = (int) Math.round(spindexerRetryOriginalPos);
                            gamepad2.rumble(0.5, 0.5, 200);
                            telemetry.addData("SPINDEXER", "JAM RETRY #" + spindexerRetryAttempts + " > FORWARD shit bro");
                        }
                    }
                }
            } else {
                spindexerMoveStartTime = 0;
                spindexerRetryAttempts = 0;
                spindexerRetryDirection = 1.0;
                spindexerLastRetryTime = 0;
            }

            if (spindexerAlignPending) {
                alignSpindexerToStep(spindexerCurrentPos);
                spindexerAlignPending = false;
            }// fuck you fuck you fuck you fuck you fuck you fuck you
        }
    }

    private void alignSpindexerToStep(double currentPos) {
        double step = Spindexer.shootingModeOffset;
        double aligned = Math.round(currentPos / step) * step;
        double offset = Math.abs(currentPos - aligned);
        if (offset > 5.0) {
            spindexerTargetPosition = (int) aligned;
            spindexerAlignPending = true;
            telemetry.addData("ALIGN", "Adjusting to " + aligned);
        }
    }
